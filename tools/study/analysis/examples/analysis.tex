\section{数学分析符号计算示例}

\subsection{极限}
\[ \lim_{x \to 0} \frac{\sin x}{x} \]

\subsection{导数}
\[ \frac{d}{dx} (\sin x \cdot e^x) \]

\subsection{偏导数}
\[ \frac{\partial^2 f}{\partial x \partial y} \]

\subsection{积分}
\[ \int \sin x \, dx \]
\[ \int_0^\infty e^{-x} \, dx \]

\subsection{泰勒展开}
\[ \sin x = x - \frac{x^3}{6} + \cdots \]

\subsection{傅里叶级数}
\[ f(x) \sim \sum_{n=1}^\infty \frac{\sin(nx)}{n} \]

\subsection{级数求和}
\[ \sum_{n=1}^\infty \frac{1}{n^2} = \frac{\pi^2}{6} \]

\subsection{常微分方程}
\[ f'(x) = f(x) \]
